# Anaconda 4.3.23
# It is recommended to install on a new environment

# CVXPY 1.0 - DCP Interface
conda install -c cvxgrp 'cvxpy<1.0'
# SDP Solvers
# Update SCS to v.2
# DO RECOMPILE - it can make a big difference in performance. 
pip install 'scs>=2.0.2' --no-binary :all:
# MOSEK 8.1.33 or higher
# Requires license in ~/mosek/mosek.lic
conda install -c mosek mosek=8.1.33
# Test solvers
conda install nose
nosetests cvxpy

# Basic libraries
conda install numpy scipy matplotlib
pip install GPy==1.8.4 numdifftools gitpython

# Library for LP-EI
# Install the following MODIFIED version of GPyOpt
# https://github.com/nrontsis/GPyOpt/tree/d3427aa464d2882c6ab12204592f187367190c18
# Run from inside the modified GPyOpt folder
pip install .

# Library for QEI
# Conda's rpy2 and R's (modified) DiceOptim
conda install -c r 'rpy2>=2.8.5'
# You might need to install the following compilers
conda install clang_osx-64 clangxx_osx-64 gfortran_osx-64
# Install modified DiceOptim version
# https://github.com/nrontsis/DiceOptim/tree/643c7f66a78f4276239125e8a4f5872bd5c30e67
# (Run: install.packages('devtools') library('devtools') build(./)
# and install(./) from R, when inside DiceOptim's folder)
